75  /*
76  * A wavelength of 1 produces no wave at all, instead the whole plane goes up and down uniformly.
77  * Other small wavelengths produce ugly waves that can even move backwards.
78  * This problem is causes by the limited resolution of our plane mesh. Because vertices are spaces one unit apart,
79  * it cannot deal with wavelengths of 2 or smaller. In general, you have to keep the wavelength greater than twice the edge
80  * length of the triangles in your mesh. You don't want to cut it too close, because waves made up of two or three quads don't look good.
81  * Either use larger wavelengths, or increase the resolution of your mesh. The simplest approach is to just use another mesh.
82  * Here is an alternative plane mesh that consists of 100×100 quads, instead of just 10×10. Each quad is still 1×1 unit,
83  * so you'll have to zoom out and multiply the wave properties by 10 to get the same result as before.
84  *

119  // We make the computational grid. Usually we only have to do this at "Awake" or when grid size changes.
120  // It eats up memory, so we are careful onlu to call it once.

122  void makeGrid() {
123  // force gridSize into a reasonable range.
124  if (gridSize <3)
125  gridSize = 3;
126  if (gridSize > 301)
127  gridSize = 301;
128  // set array sizes
129  vertices = new Vector3[gridSize*gridSize]; // gridsize in two dimentions
130  triangles = new int[gridSize * gridSize * 6]; // 6 sides for 2 triangles in each square of grid
131  // set tracker integers
132  int v = 0; // tracker for vertices
133  int t = 0; // tracker for triangles
134 
135  // we avoid offsets and scalings. This should be done in the camera, not in our computational grid.
136 
137  // Setting the vertices
138  float xmin = -boundaries;
139  float xmax = boundaries;
140  float zmin = -boundaries;
141  float zmax = boundaries;
142  float x, z;
143  for (int i = 0; i < gridSize; i++) { // iterating through the x dimension
144  x = xmin + i*(xmax - xmin)/(gridSize-1);
145  for (int j = 0; j < gridSize; j++) { // iterating through the z dimension
146  z = zmin + j*(zmax - zmin)/(gridSize-1);
147  vertices[v].x = x;
148  vertices[v].y = 0;
149  vertices[v].z = z;
150  v++;
151  }
152  }
153 
154  v = 0; // reset vertex tracker, as it got incresed in the previous for loop
155 
156  // Setting each cell's triangles
157  for (int i = 0; i < gridSize-1; i++) { // iterating through the x dimension
158  for (int j = 0; j < gridSize-1; j++) { // iterating through the z dimension - no need the equal side cause no + 1
159  // triangles are just integer indices that will reference the vertice list.
160  // each triangle points to 3 different vertices. There are six per square
161  triangles[0 + t] = 0 + v; // bottom left corner
162  triangles[1 + t] = triangles[4 + t] = 1 + v; // bottom right corner
163  triangles[2 + t] = triangles[3 + t] = gridSize + v; // top left corner
164  triangles[5 + t] = gridSize + v + 1; // top right corner
165 
166  v++;
167  t += 6; //6 because of the triangle situation
168  }
169  v++; // iterate our vertices one more time so that we start at a new row
170  }
171 
172 
173  // Now we set up the computational arrays for the FD computations, with appropriate index numbers
174  yVals = new float[3, gridSize * gridSize];
175  yPast = 0; yPresent = 1; yFuture = 2;
176 
177  // set up the initial values for the waveform
178  float dx = .1f * (1 / 30f);
179  float dz = .1f * (1 / 30f);
180  for (v = 0; v < gridSize * gridSize; v++)

183  yVals[yPresent, v] = Gauss(vertices[v].x - dx, vertices[v].z, 0f);

299  // Finite differential methodes

301  // we update the finite difference calculations by one time step
302  // The formula comes from the central difference for the 2nd derivative of Y, so
303  // y(t+1) = 2*y(t) - y(t-1) + alpha*Laplacian of y(t).
304  // The magic constant is alpha = (c*dt/dx)^2, where c = velocity, dt = time step, dx = dz = spatial step
305 
306  void oneFDstep()
307  {
308 
309  float dx = 2f / (gridSize - 1); // assume xmin, xmax spans a distance of 2.
310  float dt = 1f / 30f; // thirty frames a second, maybe
311  float c = 0.1f; // default velocity of 1/10 unit length per second
312  float alpha = (c * dt / dx) * (c * dt / dx);
313 
314  int v = gridSize; // Skip the first row, which we want to do (zero boundary)
315  for (int i = 1; i < gridSize - 1; i++)
316  {
317  for (int j = 1; j < gridSize - 1; j++)
318  {
319  v++; // move in by one column. Note we miss the first column, which we want to do anyway
320  yVals[yFuture, v] = 2f * yVals[yPresent, v] - yVals[yPast, v] - alpha * (4f * yVals[yPresent, v] - yVals[yPresent, v - 1] - yVals[yPresent, v + 1] - yVals[yPresent, v - gridSize] - yVals[yPresent, v + gridSize]);
321  }
322  v += 2; // skip over the last column as well
323  }
324  // increment the past/present/future indices
325  yPast = (++yPast) % 3;
326  yPresent = (++yPresent) % 3;
327  yFuture = (++yFuture) % 3;
328  }
329 
330  // We set the edges of the computational grid to zero. This gives a hard reflector for the wave.
331  void zeroFDedges()
332  {
333  for (int i = 0; i < gridSize; i++) // bottom edge
334  yVals[yPast, i] = yVals[yPresent, i] = yVals[yFuture, i] = 0f;
335  for (int i = 0; i < gridSize * gridSize; i += gridSize) // left edge
336  yVals[yPast, i] = yVals[yPresent, i] = yVals[yFuture, i] = 0f;
337  for (int i = (gridSize - 1) * gridSize; i < gridSize * gridSize; i++) // top edge
338  yVals[yPast, i] = yVals[yPresent, i] = yVals[yFuture, i] = 0f;
339  for (int i = gridSize - 1; i < gridSize * gridSize; i += gridSize) // right edge
340  yVals[yPast, i] = yVals[yPresent, i] = yVals[yFuture, i] = 0f;
341  }
