47  var p3 = new Vector3(Mathf.Ceil(localPos.x), 0, Mathf.Floor(localPos.z));

59 
60  //get the max distance to one of the edges and take that to compute max - dist
61  // and the it will ignore the point the furthers, hence leaving three points, to know the point is in which mesh
62  var max = Mathf.Max(Vector3.Distance(p1, localPos), Vector3.Distance(p2, localPos), Vector3.Distance(p3, localPos), Vector3.Distance(p4, localPos) + Mathf.Epsilon);
63  var dist = (max - Vector3.Distance(p1, localPos))
64  + (max - Vector3.Distance(p2, localPos))
65  + (max - Vector3.Distance(p3, localPos))
66  + (max - Vector3.Distance(p4, localPos) + Mathf.Epsilon);
67  //weighted sum
68  var height = Mesh.vertices[index(p1.x, p1.z)].y * (max - Vector3.Distance(p1, localPos))
69  + Mesh.vertices[index(p2.x, p2.z)].y * (max - Vector3.Distance(p2, localPos))
70  + Mesh.vertices[index(p3.x, p3.z)].y * (max - Vector3.Distance(p3, localPos))
71  + Mesh.vertices[index(p4.x, p4.z)].y * (max - Vector3.Distance(p4, localPos));
72 
73  //scale
74  return height * transform.lossyScale.y / dist;
